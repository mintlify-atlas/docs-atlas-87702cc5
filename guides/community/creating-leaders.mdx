This guide walks through creating a new leader from planning to implementation, with XML examples for each component.

## Prerequisites

- Familiarity with [Modding Architecture](/guides/community/modding-architecture)
- Understanding of [Creating Civilizations](/guides/community/creating-civilizations)
- Knowledge of the [Modifier System](/guides/official/modifier-system)

## Planning your leader

### Design principles

1. **Historical accuracy** — research the figure to ensure authentic representation
2. **Gameplay niche** — design abilities that offer a distinct strategic angle
3. **Civilization synergy** — abilities should complement their civilization's strengths
4. **Balance** — powerful but not dominant
5. **Specialization category** — one of: Military, Diplomacy, Expansion, Science, Culture, or Economy

### Core elements to define

- Leader name and historical description
- Core ability with clear gameplay effects
- Historical and random agendas (AI behavior)
- Compatible civilizations
- Whether the leader spans multiple Ages

## Implementation

### 1. Create the .modinfo file

Leaders are available across all Ages, so use `AlwaysMet` criteria:

```xml
<?xml version="1.0" encoding="utf-8"?>
<Mod id="your-leader-mod" version="1" xmlns="ModInfo">
  <Properties>
    <Name>Your Leader Name</Name>
    <Description>Description of your leader mod</Description>
    <Authors>Your Name</Authors>
    <AffectsSavedGames>1</AffectsSavedGames>
  </Properties>
  <Dependencies>
    <Mod id="base-standard" title="LOC_MODULE_BASE_STANDARD_NAME"/>
  </Dependencies>
  <ActionCriteria>
    <Criteria id="always"><AlwaysMet/></Criteria>
  </ActionCriteria>
  <ActionGroups>
    <ActionGroup id="main-group" scope="game" criteria="always">
      <Actions>
        <UpdateDatabase>
          <Item>leaders/your_leader.xml</Item>
          <Item>agendas/your_leader_agendas.xml</Item>
        </UpdateDatabase>
        <UpdateText>
          <Item>text/your_leader_text.xml</Item>
        </UpdateText>
      </Actions>
    </ActionGroup>
  </ActionGroups>
</Mod>
```

### 2. Define the leader

Create `leaders/your_leader.xml`:

```xml
<?xml version="1.0" encoding="utf-8"?>
<Database>
  <Types>
    <Row Type="LEADER_YOUR_LEADER" Kind="KIND_LEADER"/>
  </Types>

  <Leaders>
    <Row LeaderType="LEADER_YOUR_LEADER"
         Name="LOC_LEADER_YOUR_LEADER_NAME"
         InheritFrom="LEADER_DEFAULT"
         Description="LOC_LEADER_YOUR_LEADER_DESCRIPTION"/>
  </Leaders>

  <LeaderTraits>
    <Row LeaderType="LEADER_YOUR_LEADER" TraitType="TRAIT_LEADER_YOUR_ABILITY"/>
  </LeaderTraits>

  <LeaderCivilizations>
    <Row LeaderType="LEADER_YOUR_LEADER" CivilizationType="CIVILIZATION_YOUR_CIV"/>
  </LeaderCivilizations>

  <HistoricalAgendas>
    <Row LeaderType="LEADER_YOUR_LEADER" AgendaType="AGENDA_YOUR_HISTORICAL"/>
  </HistoricalAgendas>

  <RandomAgendas>
    <Row AgendaType="AGENDA_YOUR_RANDOM" LeaderType="LEADER_YOUR_LEADER"/>
  </RandomAgendas>
</Database>
```

### 3. Create leader abilities

Create `traits/your_leader_traits.xml`:

```xml
<?xml version="1.0" encoding="utf-8"?>
<Database>
  <Types>
    <Row Type="TRAIT_LEADER_YOUR_ABILITY" Kind="KIND_TRAIT"/>
  </Types>

  <Traits>
    <Row TraitType="TRAIT_LEADER_YOUR_ABILITY"
         Name="LOC_TRAIT_LEADER_YOUR_ABILITY_NAME"
         Description="LOC_TRAIT_LEADER_YOUR_ABILITY_DESCRIPTION"/>
  </Traits>

  <Modifiers>
    <Row ModifierId="MODIFIER_YOUR_LEADER_ABILITY_EFFECT"
         ModifierType="MODIFIER_PLAYER_ADJUST_YIELD_CHANGE"/>
  </Modifiers>

  <ModifierArguments>
    <Row ModifierId="MODIFIER_YOUR_LEADER_ABILITY_EFFECT" Name="YieldType" Value="YIELD_SCIENCE"/>
    <Row ModifierId="MODIFIER_YOUR_LEADER_ABILITY_EFFECT" Name="Amount" Value="10"/>
  </ModifierArguments>

  <TraitModifiers>
    <Row TraitType="TRAIT_LEADER_YOUR_ABILITY" ModifierId="MODIFIER_YOUR_LEADER_ABILITY_EFFECT"/>
  </TraitModifiers>
</Database>
```

### 4. Create agendas

Create `agendas/your_leader_agendas.xml`:

```xml
<?xml version="1.0" encoding="utf-8"?>
<Database>
  <Types>
    <Row Type="AGENDA_YOUR_HISTORICAL" Kind="KIND_AGENDA"/>
    <Row Type="AGENDA_YOUR_RANDOM"     Kind="KIND_AGENDA"/>
  </Types>

  <Agendas>
    <Row AgendaType="AGENDA_YOUR_HISTORICAL"
         Name="LOC_AGENDA_YOUR_HISTORICAL_NAME"
         Description="LOC_AGENDA_YOUR_HISTORICAL_DESCRIPTION"/>
    <Row AgendaType="AGENDA_YOUR_RANDOM"
         Name="LOC_AGENDA_YOUR_RANDOM_NAME"
         Description="LOC_AGENDA_YOUR_RANDOM_DESCRIPTION"/>
  </Agendas>

  <AgendaPlayerFavoredValues>
    <Row AgendaType="AGENDA_YOUR_HISTORICAL"
         ObjectiveType="OBJECTIVE_BUILD_WONDERS"
         ValueType="DIPLOVALUETYPE_POSITIVE"
         FavoredValue="10"/>
  </AgendaPlayerFavoredValues>

  <AgendaPlayerHatedValues>
    <Row AgendaType="AGENDA_YOUR_HISTORICAL"
         ObjectiveType="OBJECTIVE_RAZE_CITIES"
         ValueType="DIPLOVALUETYPE_NEGATIVE"
         HatedValue="-10"/>
  </AgendaPlayerHatedValues>
</Database>
```

## Advanced leader patterns

### Leader personas (alternate versions)

Civilization VII supports multiple personas for the same historical figure:

```xml
<Leaders>
  <Row LeaderType="LEADER_YOUR_LEADER"
       Name="LOC_LEADER_YOUR_LEADER_NAME"
       InheritFrom="LEADER_DEFAULT"/>
  <Row LeaderType="LEADER_YOUR_LEADER_ALT"
       Name="LOC_LEADER_YOUR_LEADER_ALT_NAME"
       InheritFrom="LEADER_DEFAULT"/>
</Leaders>

<LeaderTraits>
  <Row LeaderType="LEADER_YOUR_LEADER"     TraitType="TRAIT_LEADER_FIRST_PERSONA"/>
  <Row LeaderType="LEADER_YOUR_LEADER_ALT" TraitType="TRAIT_LEADER_SECOND_PERSONA"/>
</LeaderTraits>
```

### Age-specific leader abilities

Use `RequirementSets` to gate abilities by the current Age:

```xml
<Modifiers>
  <Row ModifierId="MODIFIER_LEADER_ANTIQUITY_EFFECT"
       ModifierType="MODIFIER_PLAYER_CITIES_ADJUST_BUILDING_YIELD_CHANGE"
       SubjectRequirementSetId="REQUIREMENTSET_ANTIQUITY_AGE_ACTIVE"/>
</Modifiers>

<RequirementSets>
  <Row RequirementSetId="REQUIREMENTSET_ANTIQUITY_AGE_ACTIVE"
       RequirementSetType="REQUIREMENTSET_TEST_ALL"/>
</RequirementSets>

<RequirementSetRequirements>
  <Row RequirementSetId="REQUIREMENTSET_ANTIQUITY_AGE_ACTIVE"
       RequirementId="REQUIREMENT_GAME_IN_ANTIQUITY_AGE"/>
</RequirementSetRequirements>

<Requirements>
  <Row RequirementId="REQUIREMENT_GAME_IN_ANTIQUITY_AGE"
       RequirementType="REQUIREMENT_GAME_AGE_MATCHES"/>
</Requirements>

<RequirementArguments>
  <Row RequirementId="REQUIREMENT_GAME_IN_ANTIQUITY_AGE" Name="AgeType" Value="AGE_ANTIQUITY"/>
</RequirementArguments>
```

### Cross-Age leaders

Design leaders that scale across all three Ages by linking multiple modifiers to one trait:

```xml
<Modifiers>
  <Row ModifierId="MODIFIER_LEADER_ANTIQUITY_EFFECT"
       ModifierType="MODIFIER_PLAYER_ADJUST_UNIT_PRODUCTION"
       SubjectRequirementSetId="REQUIREMENTSET_ANTIQUITY_AGE_ACTIVE"/>
  <Row ModifierId="MODIFIER_LEADER_EXPLORATION_EFFECT"
       ModifierType="MODIFIER_PLAYER_ADJUST_TRADE_ROUTE_YIELD"
       SubjectRequirementSetId="REQUIREMENTSET_EXPLORATION_AGE_ACTIVE"/>
  <Row ModifierId="MODIFIER_LEADER_MODERN_EFFECT"
       ModifierType="MODIFIER_PLAYER_ADJUST_TECHNOLOGY_SPEED"
       SubjectRequirementSetId="REQUIREMENTSET_MODERN_AGE_ACTIVE"/>
</Modifiers>

<TraitModifiers>
  <Row TraitType="TRAIT_LEADER_CROSS_AGE_ABILITY" ModifierId="MODIFIER_LEADER_ANTIQUITY_EFFECT"/>
  <Row TraitType="TRAIT_LEADER_CROSS_AGE_ABILITY" ModifierId="MODIFIER_LEADER_EXPLORATION_EFFECT"/>
  <Row TraitType="TRAIT_LEADER_CROSS_AGE_ABILITY" ModifierId="MODIFIER_LEADER_MODERN_EFFECT"/>
</TraitModifiers>
```

### Leader categories

```xml
<LeaderCategories>
  <Row LeaderType="LEADER_YOUR_LEADER" CategoryType="LEADER_CATEGORY_SCIENCE"/>
</LeaderCategories>
```

Available categories: `LEADER_CATEGORY_MILITARY`, `LEADER_CATEGORY_DIPLOMACY`, `LEADER_CATEGORY_EXPANSION`, `LEADER_CATEGORY_SCIENCE`, `LEADER_CATEGORY_CULTURE`, `LEADER_CATEGORY_ECONOMY`.

## Testing

<Steps>
  <Step title="Install and enable">
    Place the mod in `<GAME_DATA>/Mods/` and enable it.
  </Step>
  <Step title="Verify leader appears">
    Start a new game. Confirm the leader shows up with correct name and portrait.
  </Step>
  <Step title="Test abilities">
    Verify unique abilities fire correctly. Check that modifier connections work.
  </Step>
  <Step title="Test age behavior">
    Play through Age transitions. Verify age-specific abilities activate at the right times.
  </Step>
  <Step title="Test diplomacy">
    Observe AI agenda behavior and verify diplomatic responses appear.
  </Step>
</Steps>
